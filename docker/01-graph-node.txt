因alchemy提供的api性能不够作为graphql子图扫描故作废此方案
DO
$$
BEGIN
  IF NOT EXISTS (SELECT 1 FROM pg_roles WHERE rolname = 'graph') THEN
    CREATE ROLE graph LOGIN PASSWORD 'graph';
  END IF;
END
$$;

SELECT EXISTS (SELECT 1 FROM pg_database WHERE datname = 'graph_node') AS db_exists;
\gset
\if :db_exists
\else
  CREATE DATABASE graph_node OWNER graph;
\endif

ALTER DATABASE graph_node OWNER TO graph;
GRANT ALL PRIVILEGES ON DATABASE graph_node TO graph;

\connect graph_node
-- 确保 graph 用户能在 public schema 下读写
ALTER SCHEMA public OWNER TO graph;
GRANT ALL ON SCHEMA public TO graph;
GRANT CREATE ON DATABASE graph_node TO graph;

-- 初始化所需扩展
CREATE EXTENSION IF NOT EXISTS pg_stat_statements;
CREATE EXTENSION IF NOT EXISTS postgres_fdw;
GRANT USAGE ON FOREIGN DATA WRAPPER postgres_fdw TO graph;
